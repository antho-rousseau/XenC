45 // Double operations detection based on target architecture.
46 // Linux uses a 80bit wide floating point stack on x86. This induces double
47 // rounding, which in turn leads to wrong results.
48 // An easy way to test if the floating-point operations are correct is to
49 // evaluate: 89255.0/1e22. If the floating-point stack is 64 bits wide then
50 // the result is equal to 89255e-22.
51 // The best way to test this, is to create a division-function and to compare
52 // the output of the division with the expected result. (Inlining must be
53 // disabled.)
54 // On Linux,x86 89255e-22 != Div_double(89255.0/1e22)
55 #if defined(_M_X64) || defined(__x86_64__) || \
56  defined(__ARMEL__) || defined(__avr32__) || \
57  defined(__hppa__) || defined(__ia64__) || \
58  defined(__mips__) || defined(__powerpc__) || \
59  defined(__sparc__) || defined(__sparc) || defined(__s390__) || \
60  defined(__SH4__) || defined(__alpha__) || \
61  defined(_MIPS_ARCH_MIPS32R2)
62 #define DOUBLE_CONVERSION_CORRECT_DOUBLE_OPERATIONS 1
63 #elif defined(_M_IX86) || defined(__i386__) || defined(__i386)
64 #if defined(_WIN32)
65 // Windows uses a 64bit wide floating point stack.
66 #define DOUBLE_CONVERSION_CORRECT_DOUBLE_OPERATIONS 1
67 #else
68 #undef DOUBLE_CONVERSION_CORRECT_DOUBLE_OPERATIONS
69 #endif // _WIN32
70 #else
71 #error Target architecture was not detected as supported by Double-Conversion.
72 #endif

196 // Helper class for building result strings in a character buffer. The
197 // purpose of the class is to use safe operations that checks the
198 // buffer bounds on all operations in debug mode.
199 class StringBuilder {
200  public:
201  StringBuilder(char* buffer, int size)
202  : buffer_(buffer, size), position_(0) { }
203 
204  ~StringBuilder() { if (!is_finalized()) Finalize(); }
205 
206  int size() const { return buffer_.length(); }
207 
208  // Get the current position in the builder.
209  int position() const {
210  ASSERT(!is_finalized());
211  return position_;
212  }
213 
214  // Reset the position.
215  void Reset() { position_ = 0; }
216 
217  // Add a single character to the builder. It is not allowed to add
218  // 0-characters; use the Finalize() method to terminate the string
219  // instead.
220  void AddCharacter(char c) {
221  // I just extract raw data not a cstr so null is fine.
222  //ASSERT(c != '\0');
223  ASSERT(!is_finalized() && position_ < buffer_.length());
224  buffer_[position_++] = c;
225  }

227  // Add an entire string to the builder. Uses strlen() internally to
228  // compute the length of the input string.
229  void AddString(const char* s) {
230  AddSubstring(s, StrLength(s));
231  }
232 
233  // Add the first 'n' characters of the given string 's' to the
234  // builder. The input string must have enough characters.
235  void AddSubstring(const char* s, int n) {
236  ASSERT(!is_finalized() && position_ + n < buffer_.length());
237  // I just extract raw data not a cstr so null is fine.
238  //ASSERT(static_cast<size_t>(n) <= strlen(s));
239  memmove(&buffer_[position_], s, n * kCharSize);
240  position_ += n;
241  }

244  // Add character padding to the builder. If count is non-positive,
245  // nothing is added to the builder.

253  char* Finalize() {
254  ASSERT(!is_finalized() && position_ < buffer_.length());
255  buffer_[position_] = '\0';
256  // Make sure nobody managed to add a 0-character to the
257  // buffer while building the string.
258  // I just extract raw data not a cstr so null is fine.
259  //ASSERT(strlen(buffer_.start()) == static_cast<size_t>(position_));
260  position_ = -1;
261  ASSERT(is_finalized());
262  return buffer_.start();
263  }

274 // The type-based aliasing rule allows the compiler to assume that pointers of
275 // different types (for some definition of different) never alias each other.
276 // Thus the following code does not work:
277 //
278 // float f = foo();
279 // int fbits = *(int*)(&f);
280 //
281 // The compiler 'knows' that the int pointer can't refer to f since the types
282 // don't match, so the compiler may cache f in a register, leaving random data
283 // in fbits. Using C++ style casts makes no difference, however a pointer to
284 // char data is assumed to alias any other pointer. This is the 'memcpy
285 // exception'.
286 //
287 // Bit_cast uses the memcpy exception to move the bits from a variable of one
288 // type of a variable of another type. Of course the end result is likely to
289 // be implementation dependent. Most compilers (gcc-4.2 and MSVC 2005)
290 // will completely optimize BitCast away.
291 //
292 // There is an additional use for BitCast.
293 // Recent gccs will warn when they see casts that may result in breakage due to
294 // the type-based aliasing rule. If you have checked that there is no breakage
295 // you can use BitCast to cast one pointer type to another. This confuses gcc
296 // enough that it can no longer see that you have cast one pointer type to
297 // another thus avoiding the warning.
298 template <class Dest, class Source>
299 inline Dest BitCast(const Source& source) {
300  // Compile time assertion: sizeof(Dest) == sizeof(Source)
301  // A compile error here means your Dest and Source have different sizes.
302  typedef char VerifySizesAreEqual[sizeof(Dest) == sizeof(Source) ? 1 : -1]
303 #if __GNUC__ > 4 || __GNUC__ == 4 && __GNUC_MINOR__ >= 8
304  __attribute__((unused))
305 #endif
306  ;
307 
308  Dest dest;
309  memmove(&dest, &source, sizeof(dest));
310  return dest;
311 }
312 
313 template <class Dest, class Source>
314 inline Dest BitCast(Source* source) {
315  return BitCast<Dest>(reinterpret_cast<uintptr_t>(source));
316 }
